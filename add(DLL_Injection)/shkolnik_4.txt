#include <windows.h>

//https://docs.microsoft.com/en-us/windows/win32/controls/cookbook-overview
#pragma comment(linker,"\"/manifestdependency:type='win32' \
name='Microsoft.Windows.Common-Controls' version='6.0.0.0' \
processorArchitecture='*' publicKeyToken='6595b64144ccf1df' language='*'\"")


//setWindowPos - менять позицию окна в рантайме     - см MSDN
int width{}, height{};

int CALLBACK wWinMain(HINSTANCE hInstance, HINSTANCE, PWSTR crCmdLine, int nCmdShow)
{
	MSG msg{};
	HWND hwnd{};
	WNDCLASSEX wc{ sizeof(WNDCLASSEX) };
	wc.cbClsExtra = 0;
	wc.cbWndExtra - 0;
	wc.hbrBackground = reinterpret_cast<HBRUSH>(GetStockObject(WHITE_BRUSH));
	wc.hCursor = LoadCursor(nullptr, IDC_ARROW);
	wc.hIcon = LoadIcon(nullptr, IDI_APPLICATION);
	wc.hIconSm = LoadIcon(nullptr, IDI_APPLICATION);
	wc.hInstance = hInstance;
	wc.lpfnWndProc = [](HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)->LRESULT
	{
		switch (uMsg)
		{
		case WM_CREATE:
		{

			HWND hButton = CreateWindow("Button", "OK!", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 0, 0, 300, 30, hWnd, reinterpret_cast<HMENU>(1337), nullptr, nullptr);//задаём уникальный идентификатор кнопки = 1337
			//HWND hButton1 = CreateWindow("Button", "OK!", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 0, 200, 300, 30, hWnd, reinterpret_cast<HMENU>(0), nullptr, nullptr);//задаём уникальный идентификатор кнопки = 0

		}
		return 0;

		case WM_COMMAND:  //такой тип сообщения отправляется родительскому окну, когда какой-то дочерний элемент, например кнопка или иной элеиент управления был нажат/передвинут
		{
			switch (LOWORD(wParam))   //+см Button Control Notifications (MSDN)
			{
			case 1337:
			{
				HWND hCalcWnd = FindWindowW(nullptr, L"Калькулятор");//L"Безымянный - Paint"  //FindWindow
				SetWindowPos(hCalcWnd, nullptr, 0, 0, 600, 600, SWP_NOSIZE | SWP_NOOWNERZORDER | SWP_NOZORDER);//SWP_NOSIZE сообщает, что ширина и высота окна не будет изменяться, поэтому они выставлены нулями
			

				//SetWindowPos(hCalcWnd, HWND_NOTOPMOST,0,0,0,0, SWP_NOSIZE);    //обычное состояние
				//SetWindowPos(hCalcWnd, HWND_TOPMOST, 0, 0, 0, 0, SWP_NOSIZE);  //Калькулятор поверх всех окон
			}
			break;
			}
		}
		return 0;

		case WM_SIZE:
		{
			width = LOWORD(lParam);
			height = HIWORD(lParam);
		}
		return 0;

		case WM_DESTROY:
		{
			PostQuitMessage(EXIT_SUCCESS);
		}
		return 0;
		}
		return DefWindowProc(hWnd, uMsg, wParam, lParam);
	};
	wc.lpszClassName = "MyAppClass";//L"MyAppClass";
	wc.lpszMenuName = nullptr;
	wc.style = CS_VREDRAW | CS_HREDRAW;

	if (!RegisterClassEx(&wc))
		return EXIT_FAILURE;

	if (!(hwnd = CreateWindow(wc.lpszClassName, "Header", WS_OVERLAPPEDWINDOW, 0, 0, 600, 600, nullptr, nullptr, wc.hInstance, nullptr)) || hwnd == INVALID_HANDLE_VALUE)
		return EXIT_FAILURE;

	ShowWindow(hwnd, nCmdShow);
	UpdateWindow(hwnd);

	while (GetMessage(&msg, nullptr, 0, 0))
	{
		TranslateMessage(&msg);
		DispatchMessage(&msg);
	}

	return static_cast<int>(msg.wParam);

}
